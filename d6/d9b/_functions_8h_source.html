65 {
66  static std::atomic<bool> called = false;
67 
68  if (called.load())
69  return false;
70 
71  called.store (true);
72 
73  if constexpr (std::is_void_v<std::invoke_result_t<Func>>)
74  {
75  std::invoke (func);
76  }
77  else
78  {
79  if (result == nullptr)
80  std::invoke (func);
81  else
82  *result = std::invoke (func);
