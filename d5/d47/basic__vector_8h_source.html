252  void reserve (int newSize)
253  {
254  LIMES_ASSERT (newSize > 0);
255  LIMES_ASSERT (static_cast<size_type> (newSize) < std::numeric_limits<size_type>::max());

262  template <typename... Args>
263  int reserveAndInit (int newSize, Args&&... args)
264  {
265  reserve (newSize);
266 
267  auto numCreated = 0;
268 
269  for (auto i = numObjects(); i < newSize; ++i, ++numCreated)
270  objects.emplace_back (std::forward<Args> (args)...);
271 
272  return numCreated;
273  }
