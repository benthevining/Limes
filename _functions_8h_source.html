33 template <Function Func>
34 LIMES_EXPORT bool call_once (Func&& func, std::invoke_result_t<Func>* result = nullptr) noexcept (noexcept (func))
35 {
36  static std::atomic<bool> called = false;
37 
38  if (called.load())
39  return false;
40 
41  called.store (true);
42 
43  if constexpr (std::is_void_v<std::invoke_result_t<Func>>)
44  {
45  std::invoke (func);
46  }
47  else
48  {
49  if (result == nullptr)
50  std::invoke (func);
51  else
52  *result = std::invoke (func);
53  }
54 
55  return true;
56 }
