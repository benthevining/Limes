151 template <Scalar ValueType>
152 Triangle<ValueType>::Triangle (ValueType sideLength, const Angle<ValueType>& oppositeAngle, const Angle<ValueType>& anotherAngle)
153 {
154  const auto sinA = std::sin (oppositeAngle.degrees());
155 
156  const auto sideTwo = (sideLength * std::sin (anotherAngle.degrees())) / sinA;
157 
158  const auto thirdAngle = ValueType (180) - oppositeAngle.degrees() - anotherAngle.degrees();
159 
160  const auto sideThree = (sideLength * std::sin (thirdAngle)) / sinA;
161 
162  m_hypotenuse = std::max ({ sideLength, sideTwo, sideThree });
163 
164  m_base = std::min ({ sideLength, sideTwo, sideThree });
165 
166  m_height = math::middleOfThree (sideLength, sideTwo, sideThree);
167 }

169 template <Scalar ValueType>
170 Triangle<ValueType>::Triangle (ValueType sideOne, ValueType sideTwo, const Angle<ValueType>& angleBetweenThem)
171 {
172  const auto thirdSide = std::sqrt ((sideOne * sideOne) + (sideTwo * sideTwo) - (ValueType (2) * sideOne * sideTwo * std::cos (angleBetweenThem.degrees())));
173 
174  m_hypotenuse = std::max ({ sideOne, sideTwo, thirdSide });
175 
176  m_base = std::min ({ sideOne, sideTwo, thirdSide });
177 
178  m_height = math::middleOfThree (sideOne, sideTwo, thirdSide);
179 }
