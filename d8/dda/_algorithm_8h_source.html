232 template <Container Container1, Container Container2, class Callable>
233 LIMES_EXPORT constexpr int call_both (const Container1& container1, const Container2& container2, Callable&& callable)
234 {
235  int count = 0;
236 
237  for (auto first = std::begin (container1), second = std::begin (container2);
238  first != std::end (container1) && second != std::end (container2);
239  ++first, ++second, ++count)
240  {
241  if constexpr (requires { std::invoke (callable, *first, *second, count); })
242  std::invoke (callable, *first, *second, count);
243  else
244  std::invoke (callable, *first, *second);
245  }
246 
247  return count;
248 }
