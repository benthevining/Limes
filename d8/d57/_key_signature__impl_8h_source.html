27 constexpr KeySignature::KeySignature (Type typeToUse, bool isSharps, const PitchClass& pitchClassOfRoot) noexcept
28  : isFlat (! isSharps), type (typeToUse)
29 {
30  struct MajorMinorPair final
31  {
32  int numMajorAccidentals, numMinorAccidentals;
33  };
34 
35  const auto pair = [root = pitchClassOfRoot.getAsInt(), flat = isFlat]() -> MajorMinorPair
36  {
37  if (flat)
38  {
39  switch (root)
40  {
41  case (0) : return { 0, 3 };
42  case (1) : return { 6, -1 };
43  case (2) : return { -1, 1 };
44  case (3) : return { 3, 6 };
45  case (4) : LIMES_ASSERT_FALSE; return { -1, -1 };
46  case (5) : return { 1, 4 };
47  case (6) : return { 6, -1 };
48  case (7) : return { -1, 2 };
49  case (8) : return { 4, 7 };
50  case (9) : return { -1, 0 };
51  case (10) : return { 2, 5 };
52  case (11) : return { 7, -1 };
53  default : LIMES_ASSERT_FALSE; return { -1, -1 };
54  }
55  }
56 
57  switch (root)
58  {
59  case (0) : return { 0, -1 };
60  case (1) : return { 7, 4 };
61  case (2) : return { 2, -1 };
62  case (3) : return { -1, 6 };
63  case (4) : return { 4, 1 };
64  case (5) : LIMES_ASSERT_FALSE; return { -1, -1 };
65  case (6) : return { 6, 3 };
66  case (7) : return { 1, -1 };
67  case (8) : return { -1, 5 };
68  case (9) : return { 3, 0 };
69  case (10) : return { -1, 7 };
70  case (11) : return { 5, 2 };
71  default : LIMES_ASSERT_FALSE; return { -1, -1 };
72  }
73  }();
74 
75  if (type == Type::Major)
76  numAccidentals = pair.numMajorAccidentals;
77  else
78  numAccidentals = pair.numMinorAccidentals;
79 
80  LIMES_ASSERT (numAccidentals >= 0 && numAccidentals <= 7);
81 }
