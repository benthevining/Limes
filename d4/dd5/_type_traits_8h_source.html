178 {

183 
184  if (res.starts_with ("class "))

187  if (res.starts_with ("struct "))
188  return res.substr (7, res.length());

190  return res;

193 
194  if (auto* demangled = abi::__cxa_demangle (typeid (object).name(), nullptr, nullptr, &status))

199  return res;
200  }
201
