35 constexpr KeySignature::KeySignature (Type typeToUse, bool isSharps, const PitchClass& pitchClassOfRoot) noexcept
36  : isFlat (! isSharps), type (typeToUse)
37 {
38  struct MajorMinorPair final
39  {
40  int numMajorAccidentals, numMinorAccidentals;
41  };
42 
43  const auto pair = [root = pitchClassOfRoot.getAsInt(), flat = isFlat]() -> MajorMinorPair
44  {
45  if (flat)
46  {
47  switch (root)
48  {
49  case (0) : return { 0, 3 };
50  case (1) : return { 6, -1 };
51  case (2) : return { -1, 1 };
52  case (3) : return { 3, 6 };
53  case (4) : LIMES_ASSERT_FALSE; return { -1, -1 };
54  case (5) : return { 1, 4 };
55  case (6) : return { 6, -1 };
56  case (7) : return { -1, 2 };
57  case (8) : return { 4, 7 };
58  case (9) : return { -1, 0 };
59  case (10) : return { 2, 5 };
60  case (11) : return { 7, -1 };
61  default : LIMES_ASSERT_FALSE; return { -1, -1 };
62  }
63  }
64 
65  switch (root)
66  {
67  case (0) : return { 0, -1 };
68  case (1) : return { 7, 4 };
69  case (2) : return { 2, -1 };
70  case (3) : return { -1, 6 };
71  case (4) : return { 4, 1 };
72  case (5) : LIMES_ASSERT_FALSE; return { -1, -1 };
73  case (6) : return { 6, 3 };
74  case (7) : return { 1, -1 };
75  case (8) : return { -1, 5 };
76  case (9) : return { 3, 0 };
77  case (10) : return { -1, 7 };
78  case (11) : return { 5, 2 };
79  default : LIMES_ASSERT_FALSE; return { -1, -1 };
80  }
81  }();
82 
83  if (type == Type::Major)
84  numAccidentals = pair.numMajorAccidentals;
85  else
86  numAccidentals = pair.numMinorAccidentals;
87 
88  LIMES_ASSERT (numAccidentals >= 0 && numAccidentals <= 7);
89 }
