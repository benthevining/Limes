257 template <Integral T>
258 constexpr T power (T number, T power_) noexcept
259 {
260  if (power_ < T (0))
261  return T (1) / power (number, -power_);
262 
263  auto result = T (1);
264 
265  do
266  {
267  if (power_ & T (1))
268  result = result * number;
269 
270  number *= number;
271  power_ = power_ >> T (1);
272  } while (power_ > T (0));
273 
274  return result;
275 }

278 constexpr T map (T input, T sourceRangeMin, T sourceRangeMax, T targetRangeMin, T targetRangeMax) noexcept
279 {
280  LIMES_ASSERT (sourceRangeMax != sourceRangeMin); // mapping from a range of zero will produce NaN!
281  LIMES_ASSERT (targetRangeMax != targetRangeMin);
282 
283  return targetRangeMin + ((targetRangeMax - targetRangeMin) * (input - sourceRangeMin)) / (sourceRangeMax - sourceRangeMin);

304 template <Integral Integer>
305 constexpr bool isPrime (Integer number) noexcept
306 {
307  if (number <= 1)
308  return false;
309 
310  if (number == 2 || number == 3)
311  return true;
312 
313  if (isDivisibleBy (number, Integer (2)) || isDivisibleBy (number, Integer (3)))
314  return false;
315 
316  for (auto i = Integer (5); (i * i) <= number; i += 6)
317  if (isDivisibleBy (number, i) || isDivisibleBy (number, i + Integer (2)))
318  return false;
319 
320  return true;
321 }
