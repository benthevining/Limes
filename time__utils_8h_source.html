41 template <Clock InputClock, Clock OutputClock>
42 LIMES_EXPORT [[nodiscard]] Point<OutputClock> convertClockTimePoints (const Point<InputClock>& inputTime)
43 {
44  if constexpr (std::is_same_v<InputClock, OutputClock>)
45  return inputTime;
46  else
47  {
48  const auto inputNow = InputClock::now();
49  const auto systemNow = OutputClock::now();
50 
51  using OutputDuration = typename OutputClock::duration;
52 
53  const auto inputDuration = std::chrono::duration_cast<OutputDuration> (inputTime - inputNow);
54 
55  return std::chrono::time_point_cast<OutputDuration> (inputDuration + systemNow);
56  }
57 }
