122 template <Scalar DataType1, Scalar DataType2>
123 static void convert (DataType1* const dest, const DataType2* const source, int size)
124 {
125  for (auto i = 0; i < size; ++i)
126  dest[i] = static_cast<DataType1> (source[i]);
127 }
128 
129 template <Scalar DataType1, Scalar DataType2, Integral SizeType>
130 static void convert (DataType1* const dest, const DataType2* const source, SizeType size)
131 {
132  convert (dest, source, static_cast<int> (size));
133 }
