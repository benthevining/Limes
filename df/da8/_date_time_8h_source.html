131 constexpr std::tm DateTime::getStdTime() const noexcept
132 {
133  std::tm res;
134 
135  res.tm_sec = time.getSecond();
136  res.tm_min = time.getMinute();
137  res.tm_hour = time.getHour().hoursSinceMidnight();
138  res.tm_mday = date.getDayOfMonth();
139  res.tm_mon = date.getMonth().getMonthNumber();
140  res.tm_year = date.getYear().getYear() - 1900;
141  res.tm_wday = date.getWeekday<true>().daysSinceStartOfWeek();
142  res.tm_yday = date.getDayOfYear();
143 
144  // set to negative for "no information available"
145  res.tm_isdst = -1;
146 
147  return res;
