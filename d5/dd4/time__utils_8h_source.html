67 template <Clock InputClock, Clock OutputClock>
68 LIMES_EXPORT [[nodiscard]] LIMES_PURE_FUNCTION Point<OutputClock> convertClockTimePoints (const Point<InputClock>& inputTime)
69 {
70  if constexpr (std::is_same_v<InputClock, OutputClock>)
71  return inputTime;
72  else
73  {
74  const auto inputNow = InputClock::now();
75  const auto systemNow = OutputClock::now();
76 
77  using OutputDuration = typename OutputClock::duration;
78 
79  const auto inputDuration = std::chrono::duration_cast<OutputDuration> (inputTime - inputNow);
80 
81  return std::chrono::time_point_cast<OutputDuration> (inputDuration + systemNow);
82  }
83 }
