81 template <Integral T>
82 constexpr T power (T number, T power_) noexcept
83 {
84  if (power_ < T (0))
85  return T (1) / power (number, -power_);
86 
87  auto result = T (1);
88 
89  do
90  {
91  if (power_ & T (1))
92  result = result * number;
93 
94  number *= number;
95  power_ = power_ >> T (1);
96  } while (power_ > T (0));
