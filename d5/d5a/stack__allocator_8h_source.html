166 template <class T, std::size_t N, std::size_t Align>
167 template <std::size_t ReqAlign>
168 std::byte* StackAllocator<T, N, Align>::stack_buffer::allocate (std::size_t n)
169 {
170  static_assert (ReqAlign <= Align, "alignment is too small for this arena");
171 
172  LIMES_ASSERT (contains (ptr_) && "short_alloc has outlived arena");
173 
174  const auto aligned_n = align_up (n);
175 
176  if (static_cast<decltype (aligned_n)> (buf_ + N - ptr_) >= aligned_n)
177  {
178  auto* r = ptr_;
179  ptr_ += aligned_n;
180  return r;
181  }
182 
183  static_assert (Align <= alignof (std::max_align_t), "you've chosen an "
184  "alignment that is larger than alignof(std::max_align_t), and "
185  "cannot be guaranteed by normal operator new");
186 
187  return static_cast<std::byte*> (::operator new (n));
188 }

192 {
193  LIMES_ASSERT (contains (ptr_) && "short_alloc has outlived arena");
194 
195  if (contains (p))

197  n = align_up (n);

199  if (p + n == ptr_)
