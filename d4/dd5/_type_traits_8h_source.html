182 LIMES_EXPORT [[nodiscard]] std::string getDemangledTypeName (const auto& object) noexcept
183 {
184  try
185  {
186 #if LIMES_WINDOWS
187  const auto res = std::string (typeid (object).name());
188 
189  if (res.starts_with ("class "))
190  return res.substr (6, res.length());
191 
192  if (res.starts_with ("struct "))
193  return res.substr (7, res.length());
194 
195  return res;
196 #else
197  int status = 0;
198 
199  if (auto* demangled = abi::__cxa_demangle (typeid (object).name(), nullptr, nullptr, &status))
200  {
201  const auto res = std::string (demangled);
202  std::free (demangled);
203 
204  return res;
205  }
206 
207  return {};
208 #endif
209  }
210  catch (std::exception&)
211  {
212  return {};
