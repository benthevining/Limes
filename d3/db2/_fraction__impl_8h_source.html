243 template <Scalar ValueType>
244 constexpr Fraction<ValueType> Fraction<ValueType>::reduce() const noexcept
245 {
246  const auto gcd = std::max (ValueType (1), std::gcd (numerator, denominator));
247  auto res = Fraction { numerator / gcd, denominator / gcd };
248 
249  if (res.numerator < 0 && res.denominator < 0)
250  {
251  res.numerator = -res.numerator;
252  res.denominator = -res.denominator;
253  }
254 
255  return res;
256 }
