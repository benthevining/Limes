158 template <Scalar ValueType>
159 Triangle<ValueType>::Triangle (ValueType sideLength, const Angle<ValueType>& oppositeAngle, const Angle<ValueType>& anotherAngle)
160 {
161  const auto sinA = std::sin (oppositeAngle.degrees());
162 
163  const auto sideTwo = (sideLength * std::sin (anotherAngle.degrees())) / sinA;
164 
165  const auto thirdAngle = ValueType (180) - oppositeAngle.degrees() - anotherAngle.degrees();
166 
167  const auto sideThree = (sideLength * std::sin (thirdAngle)) / sinA;
168 
169  m_hypotenuse = std::max ({ sideLength, sideTwo, sideThree });
170 
171  m_base = std::min ({ sideLength, sideTwo, sideThree });
172 
173  m_height = math::middleOfThree (sideLength, sideTwo, sideThree);
174 }

176 template <Scalar ValueType>
177 Triangle<ValueType>::Triangle (ValueType sideOne, ValueType sideTwo, const Angle<ValueType>& angleBetweenThem)
178 {
179  const auto thirdSide = std::sqrt ((sideOne * sideOne) + (sideTwo * sideTwo) - (ValueType (2) * sideOne * sideTwo * std::cos (angleBetweenThem.degrees())));
180 
181  m_hypotenuse = std::max ({ sideOne, sideTwo, thirdSide });
182 
183  m_base = std::min ({ sideOne, sideTwo, thirdSide });
184 
185  m_height = math::middleOfThree (sideOne, sideTwo, thirdSide);
186 }
