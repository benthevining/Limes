164 template <Scalar ValueType>
165 Triangle<ValueType>::Triangle (ValueType sideLength, const Angle<ValueType>& oppositeAngle, const Angle<ValueType>& anotherAngle)
166 {
167  const auto sinA = std::sin (oppositeAngle.degrees());
168 
169  const auto sideTwo = (sideLength * std::sin (anotherAngle.degrees())) / sinA;
170 
171  const auto thirdAngle = ValueType (180) - oppositeAngle.degrees() - anotherAngle.degrees();
172 
173  const auto sideThree = (sideLength * std::sin (thirdAngle)) / sinA;
174 
175  m_hypotenuse = std::max ({ sideLength, sideTwo, sideThree });
176 
177  m_base = std::min ({ sideLength, sideTwo, sideThree });
178 
179  m_height = math::middleOfThree (sideLength, sideTwo, sideThree);
180 }

182 template <Scalar ValueType>
183 Triangle<ValueType>::Triangle (ValueType sideOne, ValueType sideTwo, const Angle<ValueType>& angleBetweenThem)
184 {
185  const auto thirdSide = std::sqrt ((sideOne * sideOne) + (sideTwo * sideTwo) - (ValueType (2) * sideOne * sideTwo * std::cos (angleBetweenThem.degrees())));
186 
187  m_hypotenuse = std::max ({ sideOne, sideTwo, thirdSide });
188 
189  m_base = std::min ({ sideOne, sideTwo, thirdSide });
190 
191  m_height = math::middleOfThree (sideOne, sideTwo, thirdSide);
192 }
