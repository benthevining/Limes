155 template <std::size_t N, std::size_t alignment>
156 template <std::size_t ReqAlign>
157 std::byte* stack_buffer<N, alignment>::allocate (std::size_t n)
158 {
159  static_assert (ReqAlign <= alignment, "alignment is too small for this arena");
160 
161  LIMES_ASSERT (contains (ptr_) && "short_alloc has outlived arena");
162 
163  const auto aligned_n = align_up (n);
164 
165  if (static_cast<decltype (aligned_n)> (buf_ + N - ptr_) >= aligned_n)
166  {
167  auto* r = ptr_;
168  ptr_ += aligned_n;
169  return r;
170  }
171 
172  static_assert (alignment <= alignof (std::max_align_t), "you've chosen an "
173  "alignment that is larger than alignof(std::max_align_t), and "
174  "cannot be guaranteed by normal operator new");
175 
176  return static_cast<std::byte*> (::operator new (n));
177 }

179 template <std::size_t N, std::size_t alignment>
180 void stack_buffer<N, alignment>::deallocate (std::byte* p, std::size_t n) noexcept
181 {
182  LIMES_ASSERT (contains (ptr_) && "short_alloc has outlived arena");
183 
184  if (contains (p))
185  {
186  n = align_up (n);
187 
188  if (p + n == ptr_)
189  ptr_ = p;
190  }
191  else
192  ::operator delete (p);
193 }
